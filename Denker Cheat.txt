local Players    = game:GetService("Players")

local xrayButton = xrayButton

local xrayOn   = false
local original = {}

local function setWorldTransparency(state)
	for _, obj in ipairs(workspace:GetDescendants()) do
		if obj:IsA("BasePart") then
			if state then
				original[obj] = obj.LocalTransparencyModifier
				obj.LocalTransparencyModifier = 0.5
			else
				obj.LocalTransparencyModifier = original[obj] or 0
			end
		elseif obj:IsA("Decal") then
			if state then
				original[obj] = obj.Transparency
				obj.Transparency = 0.5
			else
				obj.Transparency = original[obj] or 0
			end
		end
	end
	if not state then
		table.clear(original)
	end
end

local function setHighlights(state)
	if state then
		for _, model in ipairs(workspace:GetDescendants()) do
			if model:IsA("Model") and model:FindFirstChildWhichIsA("Humanoid") then
				if model ~= player.Character then
					if not model:FindFirstChild("XRayHighlight") then
						local hl = Instance.new("Highlight")
						hl.Name = "XRayHighlight"
						hl.FillColor = Color3.new(1, 0, 0)
						hl.FillTransparency = 0.3
						hl.OutlineColor = Color3.new(1, 1, 1)
						hl.OutlineTransparency = 0
						hl.DepthMode = Enum.HighlightDepthMode.AlwaysOnTop
						hl.Adornee = model
						hl.Parent = model
					end
				end
			end
		end
	else
		for _, model in ipairs(workspace:GetDescendants()) do
			if model:IsA("Model") then
				local hl = model:FindFirstChild("XRayHighlight")
				if hl then
					hl:Destroy()
				end
			end
		end
	end
end

	xrayOn = not xrayOn

	else
	end
end)

local guiElement = CheatIcon
local UserInputService = game:GetService("UserInputService")

local dragging = false
local dragStartPos = nil
local guiStartPos = nil
local dragThreshold = 5

local function getInputPosition(input)
	if input.UserInputType == Enum.UserInputType.MouseMovement or
		input.UserInputType == Enum.UserInputType.MouseButton1 or
		input.UserInputType == Enum.UserInputType.Touch then
		return input.Position
	end
	return nil
end

guiElement.InputBegan:Connect(function(input)
	if input.UserInputType == Enum.UserInputType.MouseButton1 or
		input.UserInputType == Enum.UserInputType.Touch then
		dragStartPos = getInputPosition(input)
		guiStartPos = guiElement.Position
		dragging = false
	end
end)

guiElement.InputChanged:Connect(function(input)
	if input.UserInputType == Enum.UserInputType.MouseMovement or
		input.UserInputType == Enum.UserInputType.Touch then
		if dragStartPos and guiStartPos then
			local currentPos = getInputPosition(input)
			if currentPos then
				local delta = currentPos - dragStartPos

				if not dragging and delta.Magnitude > dragThreshold then
					dragging = true
				end

				if dragging then
					guiElement.Position = UDim2.new(
						guiStartPos.X.Scale,
						guiStartPos.X.Offset + delta.X,
						guiStartPos.Y.Scale,
						guiStartPos.Y.Offset + delta.Y
					)
				end
			end
		end
	end
end)

guiElement.InputEnded:Connect(function(input)
	if input.UserInputType == Enum.UserInputType.MouseButton1 or
		input.UserInputType == Enum.UserInputType.Touch then
		if not dragging then
			Background.Visible = true
			BlurEffect.Size = 24
			CheatIcon.Visible = false
		end
		dragStartPos = nil
		dragging = false
		guiStartPos = nil
	end
end)

local function stopFollowing()
	following = false

	camera.CameraSubject = originalSubject or localPlayer.Character:WaitForChild("Humanoid")
	camera.CameraType = originalType or Enum.CameraType.Custom

	Spectacle3.Text = "View"
end

local function startFollowing(targetPlayer)
	if not targetPlayer.Character or not targetPlayer.Character:FindFirstChildOfClass("Humanoid") then
		Spectacle2.Text = ""
		Spectacle2.PlaceholderText = "This player is not have character or humanoid"
		task.wait(1)
		Spectacle2.PlaceholderText = "Player name"
		return
	end

	originalSubject = camera.CameraSubject
	originalType = camera.CameraType

	camera.CameraType = Enum.CameraType.Custom
	camera.CameraSubject = targetPlayer.Character:FindFirstChildOfClass("Humanoid")

	following = true
	Spectacle3.Text = "Stop"
end

Spectacle3.Activated:Connect(function()
	if following then
		stopFollowing()
	else
		local targetName = Spectacle2.Text
		local targetPlayer = Players:FindFirstChild(targetName)

		if targetPlayer then
			startFollowing(targetPlayer)
		else
			Spectacle2.Text = ""
			Spectacle2.PlaceholderText = "Player not found"
			task.wait(1)
			Spectacle2.PlaceholderText = "Player name"
		end
	end
end)

local player = game.Players.LocalPlayer
local camera = workspace.CurrentCamera
local RunService = game:GetService("RunService")

local shiftLockEnabled = false
local renderConn

local function getHumanoidParts()
	local character = player.Character or player.CharacterAdded:Wait()
	local humanoid = character:WaitForChild("Humanoid")
	local hrp = character:WaitForChild("HumanoidRootPart")
	return humanoid, hrp
end

local function rotateToCamera(hrp)
	local camLook = camera.CFrame.LookVector
	local flatLook = Vector3.new(camLook.X, 0, camLook.Z).Unit
	if flatLook.Magnitude > 0.01 then
		hrp.CFrame = CFrame.new(hrp.Position, hrp.Position + flatLook)
	end
end

local function enableShiftLock()
	shiftLockEnabled = true

	local humanoid, hrp = getHumanoidParts()
	humanoid.AutoRotate = false
	humanoid.CameraOffset = Vector3.new(1.75, 0.5, 0)

	renderConn = RunService.RenderStepped:Connect(function()
		if shiftLockEnabled then
			rotateToCamera(hrp)
		end
	end)
end

local function disableShiftLock()
	shiftLockEnabled = false
	if renderConn then renderConn:Disconnect() end

	local humanoid, _ = getHumanoidParts()
	humanoid.AutoRotate = true
	humanoid.CameraOffset = Vector3.zero
end

ShiftLock.MouseButton1Click:Connect(function()
	if shiftLockEnabled then
		disableShiftLock()
	else
		enableShiftLock()
	end
end)

player.CharacterAdded:Connect(function()
	if shiftLockEnabled then
		disableShiftLock()
		task.wait(0.1)
		enableShiftLock()
	end
end)

local CollectionService = game:GetService("CollectionService")
local Players = game:GetService("Players")

local transparencyOn = false
local originalTransparency = {}

local function setTransparency(transparency)
	for _, obj in pairs(workspace:GetDescendants()) do
		local isPlayerChar = false
		for _, player in pairs(Players:GetPlayers()) do
			if player.Character and obj:IsDescendantOf(player.Character) then
				isPlayerChar = true
				break
			end
		end

		local isNPC = CollectionService:HasTag(obj, "NPC")

		if not isPlayerChar and not isNPC and obj:IsA("BasePart") then
			if transparency == nil then
				if originalTransparency[obj] then
					obj.Transparency = originalTransparency[obj]
				end
			else
				if originalTransparency[obj] == nil then
					originalTransparency[obj] = obj.Transparency
				end
				obj.Transparency = transparency
			end
		end
	end
end

	if not transparencyOn then
		setTransparency(1)
		transparencyOn = true
	else
		setTransparency(nil)
		transparencyOn = false
	end
end)

local fireOn   = false
local fires    = {}

local function isPlayerDescendant(obj)
	for _, plr in ipairs(Players:GetPlayers()) do
		if plr.Character and obj:IsDescendantOf(plr.Character) then
			return true
		end
	end
	return false
end

local function isNPC(obj)
	return CollectionService:HasTag(obj, "NPC")
end

local function addFireToWorkspace()
	for _, part in ipairs(workspace:GetDescendants()) do
		if part:IsA("BasePart") and not isPlayerDescendant(part) and not isNPC(part) then
			if not part:FindFirstChildOfClass("Fire") then
				local fire = Instance.new("Fire")
				fire.Size  = 8
				fire.Heat  = 10
				fire.Parent = part
				table.insert(fires, fire)
			end
		end
	end
end

local function removeAllCreatedFire()
	for _, fire in ipairs(fires) do
		if fire and fire.Parent then
			fire:Destroy()
		end
	end
	fires = {}
end

FireMap.MouseButton1Click:Connect(function()
	if not fireOn then
		addFireToWorkspace()
		fireOn = true
	else
		removeAllCreatedFire()
		fireOn = false
	end
end)

local invis_on = false

	for _, part in ipairs(character:GetDescendants()) do
		if part:IsA("BasePart") then
			part.LocalTransparencyModifier = invis_on and 0.5 or 0
		end
	end
end

local function toggleInvisibility()
	invis_on = not invis_on
	local player = game.Players.LocalPlayer
	local character = player.Character or player.CharacterAdded:Wait()

	if invis_on then
		local savedpos = character.HumanoidRootPart.CFrame
		wait()
		character:MoveTo(Vector3.new(-25.95, 84, 3537.55))
		wait(0.15)

		local Seat = Instance.new('Seat', workspace)
		Seat.Anchored = false
		Seat.CanCollide = false
		Seat.Name = 'invischair'
		Seat.Transparency = 1
		Seat.Position = Vector3.new(-25.95, 84, 3537.55)

		local Weld = Instance.new("Weld", Seat)
		Weld.Part0 = Seat
		Weld.Part1 = character:FindFirstChild("Torso") or character:FindFirstChild("UpperTorso")
		wait()

		if invisChair then
			invisChair:Destroy()
		end
		InvisibleButton.Text = "Invisible"
	end
